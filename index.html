Overiew

SPARIV is aimed to be a self-contained matrix library with a simple syntax. The conversion table lists the syntax conersion for the MATLAB/Octave syntax to SPARIV. The example codes for each function are presented in a section below and the basic data types which are used in the SPARIV are available as, spariv_intt, spariv_floatt and spariv_complext. Based on these basic data types, the following user defined matrix data types are defined,

  • M_mat - double matrix

  • M_imat - long int matrix

  • M_cxmat - double complex matrix

  • M_spmat - double sparse-matrix

  • M_cxspmat - double complex sparse-matrix

Example Codes

This section of the user manual contains the example codes. Each part can be copied to lets say example.c and compile and run using,

gcc -o exec example.c -lspariv -lopenblas; ./exec

on a Linux machine. This execution style has the liberty to choose the faster blas libraries, such as openblas which is available on the user's machine. The user is also free to choose the in-built albeit slower blas library by using the following linking of the SPARIV,

gcc -o exec example.c -lsparivblas; ./exec

Core functions

■ M_new

  • initialize matrix

Usage

[O] err : if( err == 0 ){ no error returned }
[O] A   : initialize matrix 'A'

long err = M_new(     M_mat A );
long err = M_new(    M_imat A );
long err = M_new(   M_cxmat A );
long err = M_new(   M_spmat A );
long err = M_new( M_cxspmat A );

Example

#include<spariv.h>
int main()
{
    M_mat A=NULL;
    M_new( A );    // initialize matrix A

    // matrix operations

    M_delete( A );
}

■ M_zeros

  • allocate memory of dimension 'm' rows and 'n' columns for a matrix

Usage

[O] err : if( err == 0 ){ no error returned }
[I] m   : number of rows
[I] n   : number of columns
[O] A   : output matrix of size 'm' rows and 'n' columns

long err = M_zeros( long m, long n,     M_mat A );
long err = M_zeros( long m, long n,    M_imat A );
long err = M_zeros( long m, long n,   M_cxmat A );
long err = M_zeros( long m, long n,   M_spmat A );
long err = M_zeros( long m, long n, M_cxspmat A );

Example

#include<spariv.h>
int main()
{
    M_mat A=NULL;
    M_new( A );

    M_zeros( 4, 3, A );   // initialize matrix A dimension to 4 x 3
    M_disp ( A );         // print matrix A

    M_delete( A );
}


■ M_disp

  • print a spariv matrix data type or a C fundamental data type

Usage

[I] A    : input of type: M_mat, M_imat, M_cxmat, M_spmat, M_cxspmat,
                          int, long, float, double, etc.,

M_disp( [SPARIV or C - data type] A );

Example

#include<spariv.h>
int main()
{
    M_mat A=NULL;  M_new( A );

    double e = exp(1.0)

    M_zeros( 4, 3, A );       // initialize matrix A dimension to 4 x 3

    M_disp( A );
    M_disp( e );
    M_disp( "Hello world!" );

    M_delete( A );
}

■ _(A, i)

  • access the value at i-th element of 1D matrix 'A'

Usage

[I] A   : matrix
[I] i   : i-th element of matrix A  (count starts from '0')

 _(   M_mat A, long i)
 _(  M_imat A, long i)
 _( M_cxmat A, long i)

Example

#include<spariv.h>
int main()
{
    M_imat A=NULL;  M_new( A );

    M_zeros( 5, 1, A );                  // initialize matrix A as 5 x 1

    _( A, 0 ) =  1;
    _( A, 1 ) = 24;
    _( A, 3 ) = _( A, 0 ) + _( A, 1 );   // add 1 + 24, assigns 25 to _( A, 3 )

    M_disp( A );                         // prints [ 1, 24, 0, 25, 0 ]
    M_delete( A );
}


■ _(A, i,j)

  • access the value at i-th row and j-th column of matrix A

Usage

[I] A   : matrix
[I] i   : i-th row of matrix A  (count starts from '0')
[I] j   : j-th column of matrix A  (count starts from '0')

 _(   M_mat A, long i, long j)
 _(  M_imat A, long i, long j)
 _( M_cxmat A, long i, long j)

Example

#include<spariv.h>
int main()
{
    M_imat A=NULL;  M_new( A );

    M_zeros( 3,5, A );                         // initialize matrix A as 3 x 5

    _( A, 0,1 ) =  5;
    _( A, 1,4 ) = 20;
    _( A, 2,2 ) = _( A, 0,1 ) * _( A, 1,4 ); // does 5*20, assigns 100 to _( A, 2,2 )

    M_disp( A );    M_delete( A );
}


■ M_getelem

  • get the element value at i-th row and j-th column of a matrix

Usage

[I] A   : matrix
[I] i   : i-th row of matrix A  (count starts from '0')
[I] j   : j-th column of matrix A  (count starts from '0')
[O] v   : value at 'i, j' of matrix A

double         v = M_getelem(     M_mat A, long i, long j )
long           v = M_getelem(    M_imat A, long i, long j )
double complex v = M_getelem(   M_cxmat A, long i, long j )
double         v = M_getelem(   M_spmat A, long i, long j )
double complex v = M_getelem( M_cxspmat A, long i, long j )

Example

#include<spariv.h>
int main()
{
    M_imat A=NULL;  M_new( A );

    M_zeros( 3,5, A );              // initialize matrix A as 3 x 5

    _( A, 0,2 ) =  5;

    long v = M_getelem( A, 0,2 );           // 'v' stores '5'

    M_disp( v );
    M_disp( A );    M_delete( A );
}


■ M_putelem

  • put the element value at i-th row and j-th column of a matrix

Usage

[O] err : if( err == 0 ){ no error returned }
[I] A   : matrix
[I] i   : i-th row of matrix A  (count starts from '0')
[I] j   : j-th column of matrix A  (count starts from '0')
[I] v   : value at 'i, j' of matrix A

long err = M_putelem( double         v,     M_mat A, long i, long j )
long err = M_putelem( long           v,    M_imat A, long i, long j )
long err = M_putelem( double complex v,   M_cxmat A, long i, long j )
long err = M_putelem( double         v,   M_spmat A, long i, long j )
long err = M_putelem( double complex v, M_cxspmat A, long i, long j )

Example

#include<spariv.h>
int main()
{
    M_imat A=NULL;  M_new( A );

    M_zeros( 3,5, A );              // initialize matrix A as 3 x 5

    long v = 5;
    M_putelem( v, A, 0,2 );         // store the value of 'v' at (0,2)

    M_disp( A );    M_delete( A );
}

[O] err : if( err == 0 ){ no error returned }
[I] opt : set the options for subset of eigen values

    struct spariv_opts opt = {};

    opt.eigs.tol       =  0   ;    // tolerance for iteration
    opt.eigs.set_sigma = 'N'  ;    // set 'Y' to find eigen values near sigma
    opt.eigs.sigma     =  0   ;    // sigma
